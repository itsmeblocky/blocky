api reference
MethodParametersReturnsDescription
Library.new title: string, options?: { toggleKey, accentColor, icon } window creates a new UI window
Library.notify title, message, type?, duration? — shows a notification popup
Library.on eventName: string, callback: function — registers a global event listener
:addTab name: string tab adds a tab to the window
:addSection tab, label: string Frame adds a section header row
:addButton tab, label, desc?, callback? TextButton clickable button with optional description
:addToggle tab, label, desc?, default?, callback? { value, set() } checkbox toggle
:addSlider tab, label, desc?, config, callback? { value, set() } draggable slider with min/max/suffix
:addDropdown tab, label, desc?, options, default?, callback? { value, set() } dropdown selector
:addInput tab, label, desc?, placeholder?, callback? TextBox text input, fires callback on Enter
:addKeybind tab, label, desc?, default?, callback? TextButton click to bind a key, fires callback on press
:addColorPicker tab, label, desc?, default?, callback? { value, set() } hue ring + SV square picker with hex input
:addImage tab, imageId, desc?, height? ImageLabel displays a roblox asset image
:addViewport tab, model?, desc?, height? { frame, camera, setModel() } 3D viewport frame for rendering models
:addLabel tab, text: string TextLabel read-only console-style label row
:addSeparator tab Frame 1px horizontal divider line
:addConfigTab — { register(), unregister() } adds a save/load config tab
:setAccent color: Color3 — changes accent color at runtime

local Library = loadstring(game:HttpGet("https://raw.githubusercontent.com/itsmeblocky/blocky/refs/heads/main/assets/blockyExploiters.lua"))()
local ui = Library.new("My Script", { toggleKey = Enum.KeyCode.RightControl, accentColor = Color3.fromRGB(0, 120, 215), icon = "rbxassetid://0", }) local tab = ui:addTab("Main") ui:addSection(tab, "Combat") local toggle = ui:addToggle(tab, "Aimbot", false, function(state) print("aimbot:", state) end) local slider = ui:addSlider(tab, "Walk Speed", { min = 16, max = 250, default = 16, suffix = " st" }, function(v) game.Players.LocalPlayer.Character.Humanoid.WalkSpeed = v end) Library.on("buttonClicked", function(label) print("clicked:", label) end) Library.notify("Loaded", "Script ready.", "success", 3)
